# 递归

首先我们先看下递归的概念

程序调用自身的编程技巧称为递归(recursion)。

看看阶乘的例子

function factorial(n) {
    if (n == 1) return n;
    return n * factorial(n - 1)
}

console.log(factorial(5)) // 5 * 4 * 3 * 2 * 1 = 120

# 条件

  1. 当n=0, 1的时候, 结果正确.
  2. 假设函数对于n是正确的, 函数对n+1结果也正确. 如果这两点是成立的，我们知道这个函数对于所有可能的n都是正确的.


看看经典的斐波那契数列

F0 = 0 F1 = 1 Fn = Fn - 1 + Fn - 2

def fibonacci(n)
  if n == 0 then
    return 0
  elsif n == 1 then
    return 1
  else
    return fibonacci(n - 1) + fibonacci(n - 2)
  end
end

# 尾调用

尾调用，是指函数内部的最后一个动作是函数调用。该调用的返回值，直接返回给函数。


这个不是尾调用，因为 g(x) 的返回值还需要跟 1 进行计算后，f(x)才会返回值

// 非尾调用
function f(x){
    return g(x) + 1;
}

# 尾递归

上面的尾调用最后返回的函数，如果是函数自身，那就称为尾递归

我们用尾递归来优化，其实就是把所有用到的内部变量改写成函数的参数。
我们继续看开头的阶乘的例子
(PS:V8 并没有部署尾递归优化，所以其实从 Call Stack 中看不到期望的效果，也就是执行上下文栈里只有一个factorial函数）

//
function factorial(n, res) {
    if (n == 1) return res;
    return factorial(n - 1, n * res)
}

console.log(factorial(4, 1)) // 24

# 优化

function trampoline(f) {
    while (f && f instanceof Function) {
        f = f();
    }
    return f
}

function factorial(n, res) {
    if (n == 1) return res;
    return factorial.bind(null,n - 1, n * res)
}

console.log(trampoline(factorial(100,1)))

还有一个函数可以不修改原函数

function tco(f) {
  var value;
  var active = false;
  var accumulated = [];
  return function accumulator(...rest) {
    accumulated.push(rest);
    if (!active) {
      active = true;
      while (accumulated.length) {
        value = f.apply(this, accumulated.shift());
      }
      active = false;
      console.log(value);
      return value;
    }
  };
}

var factorial = tco(function factorial(n, res) {
    if(n == 1) return res;
    return factorial(n - 1, n * res)
})

console.log(factorial(100,1))
